printf <- function(str, res) {
  print(paste(str, res), quote = FALSE)
}
time_series <- wineind
time_series_length <- length(time_series)

per <- periodogram(time_series)

window_length <- 48
rank <-
  estimate.rank(
    time_series,
    L = window_length,
    method = "trmat",
    rank_range = 0:20,
    check_stat = FALSE
  )
printf("Estimated rank", rank)
## [1] Estimated rank 13
s <- ssa(time_series, L = window_length)
rec <- reconstruct(s, groups = list(1:rank))

per <- periodogram(residuals(rec))
print(per$plot + ggtitle("Periodogram of the residuals after rank detection"))
